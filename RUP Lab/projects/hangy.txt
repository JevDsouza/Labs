    /**
     * Count all matches for a char in a string.
     * @param woord
     * @param letter
     * @return count
     */
    public int countMatches(String woord, char letter) {
        int count = 0;
        for (int i=0; i < woord.length(); i++){
            if (woord.charAt(i) == letter) {
                 count++;
            }
        }
        return count;
    }
